	}

	}

				}

			add = 100
		}

			limit = { 
				OR = {
					relative_infrastructure < 0.5

					}
				}
			
			}
			add = 100
		}

			limit = { 
				OR = {
					relative_infrastructure < 0.25 
					AND = {
						has_decree = decree_road_maintenance
						relative_infrastructure < 0.5
					}
				}
			
			}
			add = 100
		}

			}
			multiply = 0

	}
}
